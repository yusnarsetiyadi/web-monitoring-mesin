=== 2. SEQUENCE DIAGRAMS ===

--- 2.1 Sequence: Lapor Kerusakan ---

@startuml sequence_lapor_kerusakan
actor "Operator" as operator
participant "KerusakanController" as controller
participant "Kerusakan Model" as kerusakanModel
participant "Perbaikan Model" as perbaikanModel

operator -> controller: Submit Laporan Kerusakan
controller -> kerusakanModel: Create Kerusakan
kerusakanModel --> controller: Kerusakan Created
controller -> perbaikanModel: Create Perbaikan
perbaikanModel --> controller: Perbaikan Created
controller --> operator: Redirect ke Daftar Kerusakan
@enduml


--- 2.2 Sequence: Assign Teknisi ke Kerusakan ---

@startuml sequence_assign_teknisi
actor "Admin" as admin
participant "KerusakanController" as controller
participant "Kerusakan Model" as kerusakanModel
participant "Perbaikan Model" as perbaikanModel

admin -> controller: Assign Teknisi ke Kerusakan
controller -> kerusakanModel: Update Status
kerusakanModel --> controller: Status Updated
controller -> perbaikanModel: Create Perbaikan
perbaikanModel --> controller: Perbaikan Created
controller --> admin: Redirect dengan Success
@enduml


--- 2.3 Sequence: Update Perbaikan ---

@startuml sequence_update_perbaikan
actor "Teknisi" as teknisi
participant "PerbaikanController" as controller
participant "Perbaikan Model" as perbaikanModel
participant "Kerusakan Model" as kerusakanModel

teknisi -> controller: Submit Update Perbaikan
controller -> perbaikanModel: Update Tindakan & Sparepart
perbaikanModel --> controller: Perbaikan Updated
controller -> kerusakanModel: Update Status ke Selesai
kerusakanModel --> controller: Status Updated
controller --> teknisi: Redirect dengan Success
@enduml


--- 2.4 Sequence: Export Laporan ---

@startuml sequence_export_laporan
actor "Admin" as admin
participant "ExportController" as controller
participant "Kerusakan Model" as kerusakanModel
participant "Exporter" as exporter

admin -> controller: Request Export Laporan
controller -> kerusakanModel: Query Kerusakan by Bulan/Tahun
kerusakanModel --> controller: Kerusakan Data
controller -> exporter: Generate File
exporter --> controller: File Generated
controller --> admin: Return Download File
@enduml


=== 3. CLASS DIAGRAMS ===

--- 3.1 Class: Domain Entities ---

@startuml class_domain
class User {
  - id: int
  - name: string
  - email: string
  - role: UserRole
  - password: string
  + kerusakan(): HasMany
  + perbaikan(): HasMany
}

class Mesin {
  - id: int
  - kode_mesin: string
  - nama_mesin: string
  - lokasi: string
  - kategori: string
  - gambar_mesin: string
  + kerusakan(): HasMany
}

class Kerusakan {
  - id: int
  - mesin_id: int
  - user_id: int
  - kode_kerusakan: string
  - deskripsi: string
  - status: StatusKerusakan
  - waktu_lapor: DateTime
  + mesin(): BelongsTo
  + user(): BelongsTo
}

class Perbaikan {
  - id: int
  - kerusakan_id: int
  - teknisi_id: int
  - tindakan: string
  - sparepart: string
  - waktu_mulai: DateTime
  - waktu_selesai: DateTime
  - catatan: string
  + kerusakan(): BelongsTo
  + teknisi(): BelongsTo
}

enum StatusKerusakan {
  Menunggu
  DalamPerbaikan
  Selesai
}

enum UserRole {
  Admin
  Operator
  Teknisi
}

User "1" --> "*" Kerusakan : melapor
User "1" --> "*" Perbaikan : melakukan
Mesin "1" --> "*" Kerusakan : memiliki
Kerusakan "1" --> "1" Perbaikan : diperbaiki
Perbaikan "1" --> "1" User : oleh
Kerusakan "1" --> "1" Mesin : pada
Kerusakan --> StatusKerusakan
User --> UserRole

note right of Kerusakan
  Kerusakan merepresentasikan
  laporan kerusakan mesin
  yang dibuat oleh operator
end note

note right of Perbaikan
  Perbaikan merepresentasikan
  tindakan perbaikan yang
  dilakukan oleh teknisi
end note
@enduml
