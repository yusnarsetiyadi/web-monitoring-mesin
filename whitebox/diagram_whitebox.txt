--- 4.3 Flowchart: Mesin Store ---

@startuml flowchart_mesin_store
start
:Auth Middleware Check;
:Verified Middleware Check;
:Validate Request;
if (Validation Fails?) then (Ya)
  :Return Validation Error;
  stop
else (Tidak)
  :Generate Next ID (Mesin::max('id') + 1);
  if (Has File gambar_mesin?) then (Ya)
    :Store Image (->store('mesin_images', 'public'));
  else (Tidak)
    :Set gambarPath = null;
  endif
  :Create Mesin Record (Mesin::create());
  :Redirect Back with Success;
  stop
endif
@enduml


--- 4.4 Flowchart: Mesin Update ---

@startuml flowchart_mesin_update
start
:Auth Middleware Check;
:Verified Middleware Check;
:Validate Request;
if (Validation Fails?) then (Ya)
  :Return Validation Error;
  stop
else (Tidak)
  :Find Mesin by ID (Mesin::findOrFail($id));
  :Prepare Update Data;
  if (Has New gambar_mesin?) then (Ya)
    if (Old Image Exists?) then (Ya)
      :Delete Old Image (Storage::disk('public')->delete());
    endif
    :Store New Image (->store('mesin_images', 'public'));
  endif
  :Update Mesin Record ($mesin->update());
  :Redirect Back with Success;
  stop
endif
@enduml

--- 4.9 Flowchart: Kerusakan Update (Assign) ---

@startuml flowchart_kerusakan_update
start
:Auth Middleware Check;
:Verified Middleware Check;
:Validate Request;
if (Validation Fails?) then (Ya)
  :Return Validation Error;
  stop
else (Tidak)
  :Begin Transaction (DB::beginTransaction());
  try {
    :Find Kerusakan by ID (Kerusakan::findOrFail($id));
    :Update Kerusakan Status;
    if (Status = "Dalam Perbaikan" AND teknisi_id exists?) then (Ya)
      :Check Existing Perbaikan (Perbaikan::where('kerusakan_id'));
      if (Perbaikan Exists?) then (Ya)
        :Rollback Transaction (DB::rollBack());
        :Redirect with Error "Perbaikan sudah ada";
        stop
      else (Tidak)
        :Create Perbaikan Record (Perbaikan::create());
      endif
    endif
    :Commit Transaction (DB::commit());
    :Redirect with Success;
    stop
  } catch (Exception) {
    :Rollback Transaction (DB::rollBack());
    :Redirect Back with Error;
    stop
  }
endif
@enduml

--- 4.11 Flowchart: Perbaikan Index ---

@startuml flowchart_perbaikan_index
start
:Auth Middleware Check;
:Verified Middleware Check;
:Get Search Query ($request->query('search'));
:Build Query (Perbaikan::with(['kerusakan', 'teknisi', 'kerusakan.mesin']));
if (User Role = Teknisi?) then (Ya)
  :Filter by teknisi_id (->where('teknisi_id', Auth::id()));
endif
if (Search Query Exists?) then (Ya)
  :Apply Search Filters (->where()->orWhere()->orWhereHas());
endif
:Execute Query (->latest()->get());
:Render Inertia View ('perbaikan/index');
stop
@enduml
